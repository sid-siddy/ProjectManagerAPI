CREATE TABLE  [dbo].[tblProject](
    ProjectId int IDENTITY(1,1) NOT NULL,
    ProjectName nvarchar(20) ,
    PStartDate Date,
PEndDate Date,
PPriority int ,
ManagerId int,
Nooftasks int,
completed int,
PStatus bit,
    PRIMARY KEY (ProjectId)   
);

CREATE TABLE [dbo].[tblTask] (
    TaskId int IDENTITY(1,1) NOT NULL,
    TaskName nvarchar(20) NOT NULL,
    TStartDate Date,
TEndDate Date,
TPriority int,
TStatus Bit, 
ParentId int,
ProjectId int, 
ParentTaskName nvarchar(20), 
UserId int,
FOREIGN KEY (ProjectId) REFERENCES tblProject(ProjectId),
FOREIGN KEY (ParentId) REFERENCES tblParent(ParentId),
PRIMARY KEY (TaskId)
);


CREATE TABLE [dbo].[tblUser] (
    UserId int IDENTITY(1,1) NOT NULL,
    FirstName nvarchar(30) NOT NULL,
    LastName nvarchar(30) NOT NULL,
EmployeeId nvarchar(10) NOT NULL,  
PRIMARY KEY (UserId)
);

CREATE TABLE [dbo].[tblParent](
[ParentId] [int] IDENTITY(1,1) NOT NULL,
[ParentTask] [nvarchar](50) NULL,
PRIMARY KEY(ParentId)
);
